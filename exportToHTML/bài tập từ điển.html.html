<h1 id="result"></h1>
<script>
    let arrayVn = ["màu đỏ", "màu đen", "ngôi nhà"]
    let arrayEng = ["red", "black","house"]
    function display() {
        let wordInput = document.getElementById("input").value
        let flag = false
        let index = 0
        for (let i = 0; i < arrayEng.length; i++) {
            if (wordInput === arrayEng[i]) {
                flag = true
                index = arrayVn[i]
                break
            }
            if (wordInput === arrayVn[i]) {
                flag = true
                index = arrayEng[i]
                break
            }
        }
        if (flag) {
            document.getElementById("result").innerHTML = "kết quả: " + index
        }else {
            document.getElementById("result").innerHTML = ""
            alert("không tìm thấy!!!")
        }
    }
</script>
